217. Contains Duplicate

class Solution {
public:
    bool containsDuplicate(vector<int>& nums) {
        unordered_map<int, int> hashmap;
        for (int i = 0; i < nums.size(); i++) {
            unordered_map<int, int>::const_iterator got = hashmap.find(nums[i]);
            if (got == hashmap.end()) {
                hashmap.insert({nums[i], i});
                continue;
            } else {
                return true;
            }
        }
        return false;
    }
};
